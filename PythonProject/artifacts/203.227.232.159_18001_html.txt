    (function () {
        'use strict';

        let callbackId = Math.floor(Math.random() * 2000000000);
        let callbacks = {};

        const _isIphone = () => {
            if ((navigator.userAgent.indexOf('iPhone') !== -1) ||
                (navigator.userAgent.indexOf('iPod') !== -1) ||
                (navigator.userAgent.indexOf('iPad') !== -1)) return true;
            else return false;
        }

        const _isAndroid = () => {
            if (navigator.userAgent.indexOf('Android') !== -1) return true;
            else return false;
        }

        window.nativeBridge = {
            exec : (successCallback, failCallback, action, param) => {
                const sCallbackId = action + (callbackId++).toString();

                callbacks[sCallbackId] = {
                    success : successCallback,
                    fail : failCallback
                }

                if (_isAndroid()) {
                    window.NativeCall.exec(sCallbackId, action, JSON.stringify(param));
                } else if (_isIphone()) {
                    let nativeParam = {
                        'header' : {
                            'callbackId' : sCallbackId,
                            'action' : action
                        },
                        'body': {
                            'param' : param
                        }
                    };

                    window.webkit.messageHandlers.nativeCall.postMessage(JSON.stringify(nativeParam));
                }
            }
        }
    })();

    const exitWebview = () => {
        nativeBridge.exec(
            (result) => { },
            (e) => { },
            'exit',
            [{ actionType: '' }]
        );
    }
