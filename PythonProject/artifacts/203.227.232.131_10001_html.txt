	function notiBoardTab(e){
	var $eTarget = $(e.currentTarget);
		var $targetPanel = $('[aria-labelledby="' + $eTarget.attr('id') + '"]');

		if(e.type === 'click') {
			$eTarget
				.attr('aria-selected', true)
				.addClass('active')
				.siblings('[role="tab"]')
				.attr('aria-selected', false)
				.removeClass('active');
			$targetPanel
				.attr('aria-hidden', false)
				.addClass('active')
				.siblings('[role="tabpanel"]')
				.attr('aria-hidden', true)
				.removeClass('active');
		} else if (e.type === 'keydown' && e.which === 13){
			$(this).click();
		}
	}

<script type="text/javascript">
	var mainSlider;
	var prdSlider;
	var notifySlider;
	var eventSlider;
	var cgSlider;

	$(function(){
		//메인슬라이드
		mainSlider = Slider();
		mainSlider.initHandle({content:'mainSlider', subMotion:true, autoPlay:true, autoPlayTime:5000});

		//추천상품 슬라이드
		prdSlider = Slider();
		prdSlider.initHandle({content:'prdSlider', subMotion:false, autoPlay:false, autoPlayTime:5000});

		//이벤트 슬라이드
		eventSlider = Slider();
		eventSlider.initHandle({content:'eventSlider', subMotion:true, autoPlay:true, autoPlayTime:5000});
		
		//알림 슬라이드
		notifySlider = Slider();
		notifySlider.initHandle({content:'notifySlider', autoPlay:false});
		
		//창공 슬라이드
		cgSlider = Slider();
		cgSlider.initHandle({content:'cgSlider', autoPlay:false});

		
		var countSlider = setInterval(function(){
			commSilder("mainSlider");
			commSilder("prdSlider");
			commSilder("eventSlider");
			commSilder("cgSlider");
		},30);

	});

	function commSilder(id){
//			var totalNo = $('#'+id+'.visual-slider li').length;
			var currentNo = $('#'+id+' .visual-slider li.active').attr('data-no');

			//$('#'+id+' .visual-slider').attr('date-total-items', totalNo);
			$('#'+id+' .visual-slider li').each(function(index, node){
				$(node).attr('data-no', index + 1);
			});
			if(currentNo == "undefined" || currentNo == null) {
				currentNo = 1;
			}
			$('#'+id+' .visual-slider-controller .count .now').html(currentNo);
			//$('#'+id+' .visual-slider-controller .count .total').html(totalNo);
	}
	/*시작 20170206 플립 김종환 통합슬라이더 추가*/
	/****************************************************************
	*
	* 통합슬라이더 20170203 플립 김종환
	*
	*****************************************************************/
	var Slider = (function() {
		var timeOut;
		var op = {
			init: false,
			content: 'visual-slider',
			slideTime: 600,
			subMotion: false,
			autoPlay: false,
			autoPlayTime: 1000,
			touching: false
		}

		var initHandle = function(option) {
			$.extend(op, option);
			op.con = $('#'+op.content);
			op.slideCon = op.con.find('.visual-slider');
			op.slide = op.con.find('.visual-slider > li');
			op.slideLen = op.slide.length;
			op.direction = op.con.find('.direction');
			op.playNum = 0;
			op.moing = false;
			op.inter = null;
			op.btnstart = op.direction.children('.slider-behavior'); 
			op.slide.eq(0).children('p').eq(0).stop().fadeIn(1000);
			op.slide.eq(0).children('p').eq(1).stop().delay(100).fadeIn(1000);
			op.slide.eq(0).children('p').eq(2).stop().delay(200).fadeIn(1000);
			op.slide.eq(0).children('a').stop().delay(300).fadeIn(1000);
			op.pager = op.con.find('.visual-slider-nav');
			op.init = true;
			op.startX = 0;
			op.startY = 0;
			op.tstate = false;
			op.mdirection = 'left';

			//pager
			var pagerEL = '';
			var bannerTxt = '';
			for(var i=0;i<op.slideLen;i++) {
				if(i == 1){
					bannerTxt = "두번째 배너";
				}else if(i == 2){
					bannerTxt = "세번째 배너";
				}else if (i == 3){
					bannerTxt = "네번째 배너";
				}else if (i == 4){
					bannerTxt = "다섯번째 배너";
				}else{
					bannerTxt = "첫번째 배너";
				}

				if(i==0) {
					pagerEL += '<a href="#" class="on"><span class="blind">'+op.slide.eq(i).find('img').eq(0).attr('alt')+'<em>'+bannerTxt+'</em></span></a>';
				}else{ 
					pagerEL += '<a href="#"><span class="blind">'+op.slide.eq(i).find('img').eq(0).attr('alt')+'<em>'+bannerTxt+'</em></span></a>';
				}
			}

			op.pager.append(pagerEL);

			op.slide.eq(0).children('a').fadeIn(600);


			if(op.slideLen<=1) return false;


			//이벤트바인딩
			eventHandle();

			//자동슬라이드
			if(op.autoPlay) autoSlideHandle();
		};

		var eventHandle = function() {
			//자동슬라이더시 오버스탑
			if(op.autoPlay) {
				op.con.off().on('mouseenter mouseleave', function(e) {
					if(e.type == 'mouseenter') {
						autoSlideStopHandle();
					}else{
						autoSlideHandle();
					}
				});

				//자동슬라이더 버튼
				op.btnstart.off().on('click', function(e) {
					e.preventDefault();
					var _this = $(this);
					if(_this.is('.slider-start')) {
						_this.removeClass('slider-start').addClass('slider-stop');
						_this.children('.blind').text('정지');
						autoSlideHandle();
					}else{
						_this.removeClass('slider-stop').addClass('slider-start');
						_this.children('.blind').text('재생');
						autoSlideStopHandle();
					}
				});
			}


			if($('html').is('.touch')) {
				op.slideCon.on('touchstart', function(e){
					//e.preventDefault();
					var orig = e.originalEvent;
					autoSlideStopHandle();
					op.startX = orig.changedTouches[0].pageX;
					op.startY = orig.changedTouches[0].pageY;
				});
				op.slideCon.on('touchmove', function(e){
					var orig = e.originalEvent;
					var moveX = orig.changedTouches[0].pageX;
					var moveY = orig.changedTouches[0].pageY;
					var xMovement = Math.abs(moveX - op.startX);
					var yMovement = Math.abs(moveY - op.startY);
					if(xMovement > yMovement){
						e.preventDefault();
						if(op.startX > moveX) op.mdirection = 'left';
						else op.mdirection = 'right';
						op.touching = true;
					}
				});
				op.slideCon.on('touchend', function(e){
					if(op.touching) {
						op.touching = false;
						autoSlideHandle();
						if(op.mdirection == 'left') {
							op.playNum++;
							if(op.playNum > op.slideLen-1) op.playNum = 0;
						}else{
							op.playNum--;
							if(op.playNum < 0) op.playNum = op.slideLen-1;
						}
						actionHandle(op.mdirection);
					}
				});
			}

			//좌우버튼
			op.direction.find('a.slider-arrow').off().on('click', function(e) {
				e.preventDefault();
				if(op.moing) return false;
				var _this = $(this);
				if(_this.is('.next')) {
					op.playNum++;
					if(op.playNum > op.slideLen-1) op.playNum = 0;
					actionHandle('left');
				}else{
					op.playNum--;
					if(op.playNum < 0) op.playNum = op.slideLen-1;
					actionHandle('right');
				}
			});

			//pager
			op.pager.find('a').off().on('click', function(e) {
				e.preventDefault();
				if(op.moing) return false; //20170307 플립 김종환 페이저 모션중 클릭방지
				op.playNum = $(this).index();
				actionHandle('left');

			});
		};
		var actionHandle = function(d) {
			if(op.moing) return false;
			var dr = 'right';
			if(d == 'right') dr = 'left';
			op.moing = true;
			op.slide.filter(':visible').stop().hide('slide', {direction:d}, function(){
				op.moing = false;
				if(op.subMotion)  $(this).children().hide();		
				$(this).removeClass('active');


			}, op.slideTime, 'easeInOutExpo').end().eq(op.playNum).stop().show('slide', {direction:dr}, function(){
					op.moing = false;
					if(op.subMotion) {
						$(this).children('p').eq(0).stop().fadeIn(1000);
						$(this).children('p').eq(1).stop().delay(100).fadeIn(1000);
						$(this).children('p').eq(2).stop().delay(200).fadeIn(1000);
						$(this).children('a').stop().delay(300).fadeIn(1000);
					}
					$(this).addClass('active');
				}, op.slideTime, 'easeInOutExpo');

			if(op.subMotion) {
				op.con.stop().animate({backgroundColor:op.slide.eq(op.playNum).data('bgcolor')}, 300);
			}
			//pager
			op.con.find('.visual-slider-nav a.on').removeClass('on').children('span');
			op.con.find('.visual-slider-nav a:eq('+op.playNum+')').addClass('on').children('span');
		};

		var autoSlideHandle = function() {
			if(!op.init) return false;
			if(!op.autoPlay) return false;
			if(op.inter != null) return false;
			if(op.btnstart.is('.slider-start')) return false;
			op.inter = setInterval(function() {
				op.playNum++;
				if(op.playNum > op.slideLen-1) op.playNum = 0;
				actionHandle('left');
			}, op.autoPlayTime);

		};
		var autoSlideStopHandle = function() {
			if(!op.init) return false;
			clearInterval(op.inter);
			op.inter = null;
		};
		return {
			initHandle : initHandle,
			autoSlideHandle : autoSlideHandle,
			autoSlideStopHandle : autoSlideStopHandle
		}
	});
	/*끝 20170206 플립 김종환*/		
</script>

			function setCookieUntilMidnight(name, value){
				var now = new Date();
				var expire = new Date();

				//오늘 자정으로 만료시간 세팅
				expire.setHours(23, 59, 59, 999);
				document.cookie = name + "=" + value + ";expires=" + expire.toUTCString() + ":path=/";
			}

				var value = "; " + document.cookie;
				var parts = value.split("; " + name + "=");

				if(parts.length === 2){
					return parts.pop().split(";").shift();
				}

	function runApp(appPath, appStorePath, client){
		var checkClient;
	
		//안드로이드, 아이폰, 아이패드 확인
		if(uAgent.indexOf('iphone') > 0  || uAgent.indexOf('ipod') > 0 || uAgent.indexOf('ipad') > 0 ){
			checkClient = "i";
		}else if(uAgent.indexOf('android') > 0){
			checkClient = "a";
		}else{
			alert("iOS, 안드로이드에서 사용할 수 있습니다.");
			return;
		}
		
		if(client == checkClient){
			if(uAgent.indexOf('iphone') > 0  || uAgent.indexOf('ipod') > 0 || uAgent.indexOf('ipad') > 0 ){
				check_app.location.href = appPath;
				setTimeout("linkIphone('"+appStorePath+"')", 1000*1);
			}else if(uAgent.indexOf('android') > 0){
				check_app.location.href = appPath;
				setTimeout("linkAndroid('"+appStorePath+"')", 1000*1);
			}else{
				
				if(client == "i"){
					alert("iOS(아이폰, 아이패드)에서만 사용할 수 있습니다.");
				}else if(client=="a"){
					alert("안드로이드OS 에서만 사용할 수 있습니다.");
				}
				
			}
		}
	}
