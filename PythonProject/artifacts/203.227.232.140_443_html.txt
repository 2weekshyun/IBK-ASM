		var fillZeroToNumber = function(str)
	    {
	        if (str < 10)
	            return '0' + str;
	        else
	            return str;
	    }
		
		var updateLeftTime = function()
	    {

	        /* 남은 시간 갱신 */
	        --timeLeft;
	
	        /* 서버 세션 타이머와 클라이언트 타이머 시간 차 보정 */
	        if (timeLeft + 0 < 0)
	        {
	            /* 타이머 해제 */
	            clearInterval(timerId);
	            /* 현재 페이지 갱신 */
	            //location.reload();
	            common_fn_pageMove("/main.sg", 'GET')
	        }
	
	        /* 남은 시간이 0이상일 경우만 표출 */
	        if (timeLeft >= 0)
	        {
	            var hours   = fillZeroToNumber(parseInt(timeLeft / 3600));
	            var minutes = fillZeroToNumber(parseInt((timeLeft % 3600) / 60));
	            var seconds = fillZeroToNumber(parseInt(timeLeft % 60));

	            /* 남은 시간 표출 */
	            $('#s_left_time').html(hours + ':' + minutes + ':' + seconds);
	        }

	function getCookie(name) {
	  var obj = name + "=";
	  var x = 0;
	  while (x <= document.cookie.length) {
	    var y = (x + obj.length);
	    if (document.cookie.substring(x, y) == obj) {
	      if ((endOfCookie = document.cookie.indexOf(";", y)) == -1)
	        endOfCookie = document.cookie.length;
	      return unescape(document.cookie.substring(y, endOfCookie));
	    }
	    x = document.cookie.indexOf(" ", x) + 1;

	    if (x == 0) break;
	  }
	  return "";
	}
	
	function popUp(ntcCnt) {
		
		for(let i=1; i<ntcCnt+1; i++){
			var cookieName = 'popupFrame'+i;
			const value = getCookie(cookieName);
			
			if(value != 'Y'){
				var option	= "";
				var width	= 550;
				var height	= 450;
				var top 	= 10 + 90*(i-1);
				var left	= 10 + 150*(i-1);
				option="width=" + width + "px, height=" + height + "px, top=" + top + "px, left=" + left + "px";
								
				window.open("popupFrame.sg?Idx="+i, "", option);
			}
		}
	}
